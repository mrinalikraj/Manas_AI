let breathInterval = null;
let breathPhase = 0;
const breathPhases = [
  { label: 'Inhale', duration: 4, scale: 1.4 },
  { label: 'Hold', duration: 4, scale: 1.4 },
  { label: 'Exhale', duration: 4, scale: 1.0 },
  { label: 'Hold', duration: 4, scale: 1.0 },
];
let breathCountdown = 4;
let breathCycles = 0;

function startBreath() {
  const btn = document.getElementById('breath-btn');
  btn.textContent = 'Stop';
  btn.onclick = stopBreath;
  breathPhase = 0; breathCycles = 0; breathCountdown = 4;
  runBreathPhase();
}

function runBreathPhase() {
  const phase = breathPhases[breathPhase];
  const circle = document.getElementById('breath-circle');
  const label = document.getElementById('breath-label');
  const count = document.getElementById('breath-count');
  if (circle) circle.style.transform = `scale(${phase.scale})`;
  if (label) label.textContent = phase.label;
  if (count) count.textContent = phase.duration;
  breathCountdown = phase.duration;
  if (breathInterval) clearInterval(breathInterval);
  breathInterval = setInterval(() => {
    breathCountdown--;
    if (count) count.textContent = breathCountdown;
    if (breathCountdown <= 0) {
      clearInterval(breathInterval);
      breathPhase = (breathPhase + 1) % 4;
      if (breathPhase === 0) breathCycles++;
      if (breathCycles >= 4) { stopBreath(); return; }
      runBreathPhase();
    }
  }, 1000);
}

function stopBreath() {
  if (breathInterval) clearInterval(breathInterval);
  const circle = document.getElementById('breath-circle');
  const label = document.getElementById('breath-label');
  const count = document.getElementById('breath-count');
  const btn = document.getElementById('breath-btn');
  if (circle) circle.style.transform = 'scale(1)';
  if (label) label.textContent = 'Done ✓';
  if (count) count.textContent = '';
  if (btn) { btn.textContent = 'Start again'; btn.onclick = startBreath; }
}
